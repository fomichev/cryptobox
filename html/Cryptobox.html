def self.stringify(obj)
  return obj.inject({}){|memo,(k,v)| memo[k.to_s] =  stringify(v); memo} if obj.is_a? Hash
  return obj.inject([]){|memo,v    | memo         << stringify(v); memo} if obj.is_a? Array
  obj
end

def self.yn(prompt)
  loop do
    print "#{prompt} [y/n]: "
    case gets.strip.downcase
    when 'y', 'yes'
      return true
    when 'n', 'no'
      return false
    end
  end
end
